	bootbox.confirm({
	title: "Add Employee",
    message: txt,

        confirm: {
            label: 'Add',
            className: 'btn-success'
        },
        cancel: {
            label: 'Cancel',
            className: 'btn-danger'
        }

		$.ajax({

  },
  dataType: 'json',

});
